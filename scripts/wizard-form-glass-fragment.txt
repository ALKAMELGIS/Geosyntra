                          value={name}

                          placeholder="Full name"
                          required

                    </label>
                    <label className="home-wizard-form__label--role home-wizard-glass-field">
                      <span className="home-wizard-glass-field__label">Role</span>

                        value={roleSlug}

                      placeholder="Email"
                      required
                    />

                </label>

                      autoComplete={mode === 'signup' ? 'new-password' : 'current-password'}
                      placeholder="Password"
                      required
                    />
                    <button
                      type="button"
                      className="home-wizard-auth-chip home-wizard-auth-chip--in-field"

                      aria-label={showPassword ? 'Hide password' : 'Show password'}

                    <GlassToggle

                      checked={keepSignedIn}
                      onChange={onKeepSignedInChange}

                      hint="Stay signed in on this device · closes when browser ends if off"
                      disabled={busy}
                    />
                  </div>
                ) : null}

                {info ? <p className="home-wizard-form__info">{info}</p> : null}
                {error ? <p className="home-wizard-form__error">{error}</p> : null}
                <SaasButton
                  size="lg"
                  variant="primary"
                  className="home-wizard-form__submit home-wizard-glass-submit"
                  onClick={() => void submit()}
                >
                  {busy ? 'Please wait…' : mode === 'signup' ? 'Create account' : 'Sign in'}
                </SaasButton>

                {mode === 'signin' ? (
                  <div className="home-wizard-forgot-row">
                    <button type="button" className="home-wizard-forgot-link" onClick={onForgotUsername}>
                      Forgot username?

                    <span className="home-wizard-forgot-sep">or</span>

                      Forgot password?
                    </button>
                  </div>
                ) : null}
